81 void USB_Check_Rx()
82 {
83  /* If the USB device isn't configured yet, we can't really do anything
84  * else since we don't have a host to talk to. So jump back to the
85  * top of the while loop. */
86  if( USBGetDeviceState() < CONFIGURED_STATE )
87  {
88  return;
89  }
90 
91  /* If we are currently suspended, then we need to see if we need to
92  * issue a remote wakeup. In either case, we shouldn't process any
93  * keyboard commands since we aren't currently communicating to the host
94  * thus just continue back to the start of the while loop. */
95  if( USBIsDeviceSuspended()== true )
96  {
97  return;
98  }
99 
100  //Check if we have received an OUT data packet from the host
101  if(HIDRxHandleBusy(USBOutHandle) == false)
102  {
103  ParsePacketIn();
104  //Always respond to the host
105  USBInHandle = HIDTxPacket(CUSTOM_DEVICE_HID_EP, (uint8_t *)&ToSendDataBuffer[0], 64);
106  //Re-arm the OUT endpoint, so we can receive the next OUT data packet
107  //that the host may try to send us.
108  USBOutHandle = HIDRxPacket(CUSTOM_DEVICE_HID_EP, (uint8_t*)&ReceivedDataBuffer[0], 64);
109  }
110 }
