114 /* Data Transfer Flags
115 
116 The following flags are used in the flags parameter of the "USBDEVTransferData"
117 and "USBHALTransferData" routines. They can be accessed by the bitfield
118 definitions or the macros can be OR'd together to identify the endpoint number
119 and properties of the data transfer.
120 
121 <code>
122  7 6 5 4 3 2 1 0 - Field name
123  | | | | \_____/
124  | | | | +----- ep_num - Endpoint number
125  | | | +---------- zero_pkt - End transfer with short or zero-sized packet
126  | | +------------ dts - 0=DATA0 packet, 1=DATA1 packet
127  | +-------------- force_dts - Force data toggle sync to match dts field
128  +---------------- direction - Transfer direction: 0=Receive, 1=Transmit
129 </code>
130 */
131

140  uint8_t force_dts: 1;
141  uint8_t direction: 1;
142  }field;
143 
144 } TRANSFER_FLAGS;

426 /*******************************************************************************
427  Function:
428  bool <Event-handling Function Name> ( USB_EVENT event,
429  void *data, unsigned int size )
430 
431  Description:
432  This routine is a "call out" routine that must be implemented by
433  any layer of the USB SW Stack (except the HAL which is at the
434  root of the event-call tree that needs to receive events. When
435  an event occurs, the HAL calls the next higher layer in the
436  stack to handle the event. Each layer either handles the event
437  or calls the layer above it to handle the event. Events are
438  identified by the "event" parameter and may have associated
439  data. If the higher layer was able to handle the event, it
440  should return true. If not, it should return false.
441 
442  Preconditions:
443  USBInitialize must have been called to initialize the USB SW
444  Stack.
445 
446  Parameters:
447  USB_EVENT event - Identifies the bus event that occurred
448  void *data - Pointer to event-specific data
449  unsigned int size - Size of the event-specific data
450 
451  Return Values:
452  None
453 
454  Remarks:
455  The function is name is defined by the layer that implements
456  it. A pointer to the function will be placed by into a table
457  that the lower-layer will use to call it. This requires the
458  function to use a specific call "signature" (return data type
459  and values and data parameter types and values).
460 
461 *******************************************************************************/
