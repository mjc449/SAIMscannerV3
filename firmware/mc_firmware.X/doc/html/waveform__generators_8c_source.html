58 void WriteWord(uint16_t x_msg, uint16_t y_msg)
59 {
60  int i;
61  WV_SYNC = 0;
62  for(i = 15; i > -1 ; i--)
63  {
64  WV_SCK = 1;
65  WV_X_DATA = (x_msg >> i) & 1;
66  WV_Y_DATA = (y_msg >> i) & 1;
67  WV_SCK = 0;
68  }
69  WV_SYNC = 1;
70  WV_SCK = 1;
71 }
