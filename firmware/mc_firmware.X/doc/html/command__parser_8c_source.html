53 void ParsePacketIn(void) {
54  //Disable user interrupts before processing the packet
55  SET_CPU_IPL(7);
56  //Storage for return (error) codes, not all cases need returns
57  uint8_t response = 0;
58 
59  switch (ReceivedDataBuffer[0]) {
60  /*******************************************************************
61  * Timer functions
62  ******************************************************************/
63  case COMMAND_TIMER_ON_OFF:
64  EnableDisableTimer(dataIn[0], dataIn[1], dataIn[2]);
65  break;
66 
67  case COMMAND_STROBE_PIN:
68  StrobeOnePin(dataIn[0], dataIn[1], Make32(dataIn[2], dataIn[3], dataIn[4], dataIn[5]), Make16(dataIn[6], dataIn[7]));
69  break;
70 
71  /*******************************************************************
72  * DIO functions
73  ******************************************************************/
74  case COMMAND_SET_PIN_DIRECTION:
75  DIO_SetPinDirection(dataIn[0], dataIn[1]);
76  break;
77 
78  case COMMAND_TOGGLE_PIN:
79  DIO_TogglePin(dataIn[0]);
80  break;
81 
82  case COMMAND_WRITE_PIN:
83  DIO_WritePin(dataIn[0], dataIn[1]);
84  break;
85 
86  case COMMAND_WRITE_PORT:
87  DIO_WritePort(dataIn[0], dataIn[1]);
88  break;
89 
90  case COMMAND_READ_PORT:
91  dataOut[0] = DIO_ReadPort(dataIn[0]);
92  break;
93 
94  /*******************************************************************
95  * External trigger functions
96  ******************************************************************/
97  case COMMAND_TRIGGER_ON_OFF:
98  EnableDisableTrigger(dataIn[0], dataIn[1], dataIn[2], dataIn[3]);
99  break;
100  case COMMAND_TOGGLE_PINS_ON_TRIGGER:
101  TogglePinsOnTrigger(dataIn[0], dataIn[1]);
102  break;
103  case COMMAND_START_TIMER_ON_TRIGGER:
104  break;
105  case COMMAND_STOP_TIMER_ON_TRIGGER:
106  break;
107  case COMMAND_TOGGLE_TIMER_ON_TRIGGER:
108  ToggleTimerOnTrigger(dataIn[0], dataIn[1]);
109 
110  /*******************************************************************
111  * Waveform functions
112  ******************************************************************/
113  case COMMAND_WAVEFORM_ON_OFF:
114  WaveformOutputEnable(dataIn[0], dataIn[1]);
115  break;
116  case COMMAND_WAVEFORM_FREQUENCY:
117  SetSameFreq(Make32(dataIn[0], dataIn[1], dataIn[2], dataIn[3]));
118  break;
119  case COMMAND_WAVEFORM_PHASE:
120  SetPhase(Make16(dataIn[0], dataIn[1]));
121  break;
122  case COMMAND_WAVEFORM_OUTPUT_TYPE:
123  SetOutputTypes(dataIn[0], dataIn[1]);
124  break;
125  case COMMAND_WAVEFORM_X_FREQUENCY:
126  SetXFreq(Make32(dataIn[0], dataIn[1], dataIn[2], dataIn[3]));
127  break;
128  case COMMAND_WAVEFORM_Y_FREQUENCY:
129  SetYFreq(Make32(dataIn[0], dataIn[1], dataIn[2], dataIn[3]));
130  break;
131 
132  /*******************************************************************
133  * Special Functions
134  ******************************************************************/
135  case COMMAND_IDENTIFY_DEVICE:
136  dataOut[0] = 1;
137  break;
138 
139  case COMMAND_READ_BOOTLOADER_KEY:
140  ReadBootloaderFlag(&dataOut[0]);
141  break;
142  case COMMAND_GOTO_BOOTLOADER:
143  SetBootloaderFlag();
144  break;
145  case COMMAND_RESET_DEVICE:
146  __asm__ volatile("reset");
147  break;
148 
149  default:
150  break;
151  }
152  ToSendDataBuffer[0] = response;
153  //Keep the CPU priority low when not doing parsing a command
154  SET_CPU_IPL(3);
155 }

160 void SetBootloaderFlag(void) {
161  //The last address in the user program memory is where we'll store the flag
162  uint16_t memOffset = 0x0000;
163  uint32_t progMemAddress = 0x2A7FE;
164  //Set the very last bit to 0
165  uint16_t blFlagL = 0xFFFE;
166  uint8_t blFlagH = 0xFF;
167  //Initialize word programming and memory pointer
168  NVMCON = 0x4003;
169  TBLPAG = progMemAddress >> 16;
170  memOffset = progMemAddress & 0xFFFF;
171 
172  //Write to the latches
173  __builtin_tblwtl(memOffset, blFlagL);
174  __builtin_tblwth(memOffset, blFlagH);
175  __builtin_disi(5);
176  __builtin_write_NVM();
177  while (NVMCONbits.WR);
178  __asm__ volatile("reset");
179 }
