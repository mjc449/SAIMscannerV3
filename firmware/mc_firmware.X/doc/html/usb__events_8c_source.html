30  * Function: bool USER_USB_CALLBACK_EVENT_HANDLER(

33  * PreCondition: None
34  *
35  * Input: USB_EVENT event - the type of event

38  *
39  * Output: None
40  *
41  * Side Effects: None
42  *
43  * Overview: This function is called from the USB stack to
44  * notify a user application that a USB event
45  * occured. This callback is in interrupt context
46  * when the USB_INTERRUPT option is selected.
47  *
48  * Note: None
49  *******************************************************************/
50 bool USER_USB_CALLBACK_EVENT_HANDLER(USB_EVENT event, void *pdata, uint16_t size)
51 {
52  switch((int)event)
53  {
54  case EVENT_TRANSFER:
55  break;
56 
57  case EVENT_SOF:
58  break;
59 
60  case EVENT_SUSPEND:
61 
62  //Call the hardware platform specific handler for suspend events for
63  //possible further action (like optionally going reconfiguring the application
64  //for lower power states and going to sleep during the suspend event). This
65  //would normally be done in USB compliant bus powered applications, although
66  //no further processing is needed for purely self powered applications that
67  //don't consume power from the host.
68  SYSTEM_Initialize(SYSTEM_STATE_USB_SUSPEND);
69  break;
70 
71  case EVENT_RESUME:
72 
73  //Call the hardware platform specific resume from suspend handler (ex: to
74  //restore I/O pins to higher power states if they were changed during the
75  //preceding SYSTEM_Initialize(SYSTEM_STATE_USB_SUSPEND) call at the start
76  //of the suspend condition.
77  SYSTEM_Initialize(SYSTEM_STATE_USB_RESUME);
78  break;
79 
80  case EVENT_CONFIGURED:
81  /* When the device is configured, we can (re)initialize the demo
82  * code. */
83  APP_DeviceCustomHIDInitialize();
84  break;
85 
86  case EVENT_SET_DESCRIPTOR:
87  break;
88 
89  case EVENT_EP0_REQUEST:
90  /* We have received a non-standard USB request. The HID driver
91  * needs to check to see if the request was for it. */
92  USBCheckHIDRequest();
93  break;
94 
95  case EVENT_BUS_ERROR:
96  break;
97 
98  case EVENT_TRANSFER_TERMINATED:
99  break;
100 
101  default:
102  break;
103  }
104  return true;
105 }
