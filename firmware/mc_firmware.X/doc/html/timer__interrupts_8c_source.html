44 {
45  switch(timer)

47  case TIMER_0:
48  IFS0bits.T1IF = 0;
49  if(onOff)
50  {
51  PR1 = (uint16_t)(TIMER_0_PARAMS.period);
52  TMR1 = 0;
53  IPC0bits.T1IP = priority;
54  ENABLE_TIMER_1();
