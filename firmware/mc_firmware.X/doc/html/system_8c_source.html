68 {
69  PIN_MANAGER_Initialize();
70  INTERRUPT_Initialize();
71  OSCILLATOR_Initialize();
72  EXT_INT_Initialize();
73  TIMERS_Initialize();
74  InitializeWaveformDACs();

77  switch(state)
78  {
79  case SYSTEM_STATE_USB_START:
80  //Make sure that the general purpose output driver multiplexed with
81  //the VBUS pin is always consistently configured to be tri-stated in
82  //USB applications, so as to avoid any possible contention with the host.
83  //(ex: maintain TRISFbits.TRISF7 = 1 at all times).
84  TRISFbits.TRISF7 = 1;
85  break;
86 
87  case SYSTEM_STATE_USB_SUSPEND:
88  //If developing a bus powered USB device that needs to be USB compliant,
89  //insert code here to reduce the I/O pin and microcontroller power consumption,
90  //so that the total current is <2.5mA from the USB host's VBUS supply.
91  //If developing a self powered application (or a bus powered device where
92  //official USB compliance isn't critical), nothing strictly needs
93  //to be done during USB suspend.
94 
95  USBSleepOnSuspend();
96  break;
97 
98  case SYSTEM_STATE_USB_RESUME:
99  break;
100 
101  default:
102  break;
103  }
104 }

178 void OSCILLATOR_Initialize(void)
179 {
180  // CPDIV 1:1; RCDIV FRC/2; DOZE 1:8; DOZEN disabled; ROI disabled;
181  CLKDIV = 0x3100;
182  // TUN Center frequency;
183  OSCTUN = 0x0000;
184  // CF no clock failure; NOSC PRIPLL; SOSCEN disabled; POSCEN disabled; CLKLOCK unlocked; OSWEN Switch is Complete;
185  __builtin_write_OSCCONH((uint8_t) ((0x0300 >> _OSCCON_NOSC_POSITION) & 0x00FF));
186  __builtin_write_OSCCONL((uint8_t) ((0x0300 | _OSCCON_OSWEN_MASK) & 0xFF));
187  // Wait for Clock switch to occur
188  while (OSCCONbits.OSWEN != 0);
189 }
