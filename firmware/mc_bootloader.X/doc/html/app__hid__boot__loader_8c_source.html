179  * Function: void ProcessIO(void)
180  *
181  * PreCondition: None
182  *
183  * Input: None
184  *
185  * Output: None
186  *
187  * Side Effects: None
188  *
189  * Overview: This function is a place holder for other user routines.
190  * It is a mixture of both USB and non-USB tasks.
191  *
192  * Note: None

194 void APP_HIDBootLoaderTasks(void)
195 {
196  unsigned char i;
197  unsigned int j;
198  uint32_t_VAL FlashMemoryValue;
199 
200  if(BootState == IdleState)
201  {
202  //Are we done sending the last response. We need to be before we
203  // receive the next command because we clear the PacketToPC buffer
204  // once we receive a command
205  if(!USBHandleBusy(USBInHandle))
206  {
207  if(!USBHandleBusy(USBOutHandle)) //Did we receive a command?
208  {
209  for(i = 0; i < TotalPacketSize; i++)
210  {
211  PacketFromPC.Contents[i] = PacketFromPCBuffer.Contents[i];
212  }
213 
214  USBOutHandle = USBRxOnePacket(CUSTOM_DEVICE_HID_EP,(uint8_t*)&PacketFromPCBuffer,64);
215  BootState = NotIdleState;
216 
217  //Prepare the next packet we will send to the host, by initializing the entire packet to 0x00.
218  for(i = 0; i < TotalPacketSize; i++)
219  {
220  //This saves code space, since we don't have to do it independently in the QUERY_DEVICE and GET_DATA cases.
221  PacketToPC.Contents[i] = 0;
222  }
223  }
224  }
225  }
226  else //(BootState must be in NotIdleState)
227  {
228  switch(PacketFromPC.Command)
229  {
230  case QUERY_DEVICE:
231  {
232  //Prepare a response packet, which lets the PC software know about the memory ranges of this device.
233  PacketToPC.Command = (unsigned char)QUERY_DEVICE;
234  PacketToPC.PacketDataFieldSize = (unsigned char)RequestDataBlockSize;
235  PacketToPC.BytesPerAddress = (unsigned char)BytesPerFlashAddress;
236 
237  PacketToPC.Type1 = (unsigned char)TypeProgramMemory;
238  PacketToPC.Address1 = (unsigned long)BOOT_CONFIG_USER_MEMORY_START_ADDRESS;
239  PacketToPC.Length1 = (unsigned long)(ProgramMemStopAddress - BOOT_CONFIG_USER_MEMORY_START_ADDRESS); //Size of program memory area
240  PacketToPC.Type2 = (unsigned char)TypeEndOfTypeList;
241 
242  if(ConfigsProtected == UNLOCKCONFIG)
243  {
244  PacketToPC.Type2 = (unsigned char)TypeConfigWords;
245  PacketToPC.Address2 = (unsigned long)BOOT_MEMORY_CONFIG_START_ADDRESS;
246  PacketToPC.Length2 = (unsigned long)(BOOT_MEMORY_CONFIG_END_ADDRESS - BOOT_MEMORY_CONFIG_START_ADDRESS);
247  PacketToPC.Type3 = (unsigned char)TypeEndOfTypeList;
248  }
249 
250  //Init pad bytes to 0x00... Already done after we received the QUERY_DEVICE command (just after calling HIDRxPacket()).
251 
252  if(!USBHandleBusy(USBInHandle))
253  {
254  USBInHandle = USBTxOnePacket(CUSTOM_DEVICE_HID_EP,(uint8_t*)&PacketToPC,64);
255  BootState = IdleState;
256  }
257  break;
258  }
259 
260  case UNLOCK_CONFIG:
261  {
262  if(PacketFromPC.LockValue == UNLOCKCONFIG)
263  {
264  MaxPageToErase = BOOT_CONFIG_USER_MEMORY_ERASE_PAGE_END_CONFIGS; //Assume we will not allow erase/programming of config words (unless host sends override command)
265  ProgramMemStopAddress = BOOT_CONFIG_USER_MEMORY_END_ADDRESS_CONFIGS;
266  ConfigsProtected = UNLOCKCONFIG;
267  }
268  else //LockValue must be == LOCKCONFIG
269  {
270  MaxPageToErase = BOOT_CONFIG_USER_MEMORY_ERASE_PAGE_END_NO_CONFIGS;
271  ProgramMemStopAddress = BOOT_CONFIG_USER_MEMORY_END_ADDRESS_NO_CONFIGS;
272  ConfigsProtected = LOCKCONFIG;
273  }
274 
275  BootState = IdleState;
276  break;
277  }
278 
279  case ERASE_DEVICE:
280  {
281  for(ErasePageTracker = BOOT_CONFIG_USER_MEMORY_ERASE_PAGE_START; ErasePageTracker < (MaxPageToErase + 1); ErasePageTracker++)
282  {
283  EraseFlash();
284  USBDeviceTasks(); //Call USBDriverService() periodically to prevent falling off the bus if any SETUP packets should happen to arrive.
285  }
286 
287  NVMCONbits.WREN = 0; //Good practice to clear WREN bit anytime we are not expecting to do erase/write operations, further reducing probability of accidental activation.
288  BootState = IdleState;
289  break;
290  }
291 
292  case PROGRAM_DEVICE:
293  {
294  if(ProgrammedPointer == (unsigned long)InvalidAddress)
295  {
296  ProgrammedPointer = PacketFromPC.Address;
297  }
298 
299  if(ProgrammedPointer == (unsigned long)PacketFromPC.Address)
300  {
301  for(i = 0; i < (PacketFromPC.Size/WORDSIZE); i++)
302  {
303  unsigned int index;
304 
305  index = (RequestDataBlockSize-PacketFromPC.Size)/WORDSIZE+i;
306  ProgrammingBuffer[BufferedDataIndex] = PacketFromPC.Data[(RequestDataBlockSize-PacketFromPC.Size)/WORDSIZE+i]; //Data field is right justified. Need to put it in the buffer left justified.
307  BufferedDataIndex++;
308  ProgrammedPointer++;
309  if(BufferedDataIndex == (RequestDataBlockSize/WORDSIZE)) //Need to make sure it doesn't call WriteFlashSubBlock() unless BufferedDataIndex/2 is an integer
310  {
311  WriteFlashSubBlock();
312  }
313  }
314  }
315  //else host sent us a non-contiguous packet address... to make this firmware simpler, host should not do this without sending a PROGRAM_COMPLETE command in between program sections.
316  BootState = IdleState;
317  break;
318  }
319 
320  case PROGRAM_COMPLETE:
321  {
322  WriteFlashSubBlock();
323  ProgrammedPointer = InvalidAddress; //Reinitialize pointer to an invalid range, so we know the next PROGRAM_DEVICE will be the start address of a contiguous section.
324  BootState = IdleState;
325  break;
326  }
327 
328  case GET_DATA:
329  {
330  if(!USBHandleBusy(USBInHandle))
331  {
332  //Init pad bytes to 0x00... Already done after we received the QUERY_DEVICE command (just after calling HIDRxReport()).
333  PacketToPC.Command = GET_DATA;
334  PacketToPC.Address = PacketFromPC.Address;
335  PacketToPC.Size = PacketFromPC.Size;
336 
337  for(i = 0; i < (PacketFromPC.Size/2); i=i+2)
338  {
339  FlashMemoryValue.Val = ReadProgramMemory(PacketFromPC.Address + i);
340  PacketToPC.Data[RequestDataBlockSize/WORDSIZE + i - PacketFromPC.Size/WORDSIZE] = FlashMemoryValue.word.LW; //Low word, pure 16-bits of real data
341  FlashMemoryValue.byte.MB = 0x00; //Set the "phantom byte" = 0x00, since this is what is in the .HEX file generatd by MPLAB.
342  //Needs to be 0x00 so as to match, and successfully verify, even though the actual table read yeilded 0xFF for this phantom byte.
343  PacketToPC.Data[RequestDataBlockSize/WORDSIZE + i + 1 - PacketFromPC.Size/WORDSIZE] = FlashMemoryValue.word.HW; //Upper word, which contains the phantom byte
344  }
345 
346  USBInHandle = USBTxOnePacket(CUSTOM_DEVICE_HID_EP,(uint8_t*)&PacketToPC.Contents[0],64);
347  BootState = IdleState;
348  }
349  break;
350  }
351 
352 
353  case RESET_DEVICE:
354  {
355  U1CON = 0x0000; //Disable USB module
356  //And wait awhile for the USB cable capacitance to discharge down to disconnected (SE0) state.
357  //Otherwise host might not realize we disconnected/reconnected when we do the reset.
358  //A basic for() loop decrementing a 16 bit number would be simpler, but seems to take more code space for
359  //a given delay. So do this instead:
360  for(j = 0; j < 0xFFFF; j++)
361  {
362  Nop();
363  }
364  asm("reset");
365  break;
366  }
367  }//End switch
368  }//End if/else
369 
370 }//End ProcessIO()

374 void EraseFlash(void)
375 {
376  uint32_t_VAL MemAddressToErase = {0x00000000};
377  MemAddressToErase.Val = (((uint32_t)ErasePageTracker) * BOOT_ERASE_BLOCK_SIZE);
378 
379  NVMCON = 0x4042; //Erase page on next WR
380 
381  TBLPAG = MemAddressToErase.byte.UB;
382  __builtin_tblwtl(MemAddressToErase.word.LW, 0xFFFF);
383 
384  asm("DISI #16"); //Disable interrupts for next few instructions for unlock sequence
385  __builtin_write_NVM();
386  while(NVMCONbits.WR == 1){}
387 
388 // EECON1bits.WREN = 0; //Good practice now to clear the WREN bit, as further protection against any future accidental activation of self write/erase operations.
389 }

392 void WriteFlashSubBlock(void) //Use word writes to write code chunks less than a full 64 byte block size.
393 {
394  unsigned int i = 0;
395  uint32_t_VAL Address;
396 
397  NVMCON = 0x4003; //Perform WORD write next time WR gets set = 1.
398 
399  while(BufferedDataIndex > 0) //While data is still in the buffer.
400  {
401  Address.Val = ProgrammedPointer - BufferedDataIndex;
402  TBLPAG = Address.word.HW;
403 
404  __builtin_tblwtl(Address.word.LW, ProgrammingBuffer[i]); //Write the low word to the latch
405  __builtin_tblwth(Address.word.LW, ProgrammingBuffer[i + 1]); //Write the high word to the latch (8 bits of data + 8 bits of "phantom data")
406  i = i + 2;
407 
408  asm("DISI #16"); //Disable interrupts for next few instructions for unlock sequence
409  __builtin_write_NVM();
410  while(NVMCONbits.WR == 1){}
411 
412  BufferedDataIndex = BufferedDataIndex - 2; //Used up 2 (16-bit) words from the buffer.
413  }
414 
415  NVMCONbits.WREN = 0; //Good practice to clear WREN bit anytime we are not expecting to do erase/write operations, further reducing probability of accidental activation.
416 }

419 /*********************************************************************
420  * Function: uint32_t ReadProgramMemory(uint32_t address)
421  *
422  * PreCondition: None
423  *
424  * Input: Program memory address to read from. Should be
425  * an even number.
426  *
427  * Output: Program word at the specified address. For the
428  * PIC24, dsPIC, etc. which have a 24 bit program
429  * word size, the upper byte is 0x00.
430  *
431  * Side Effects: None
432  *
433  * Overview: Modifies and restores TBLPAG. Make sure that if
434  * using interrupts and the PSV feature of the CPU
435  * in an ISR that the TBLPAG register is preloaded
436  * with the correct value (rather than assuming
437  * TBLPAG is always pointing to the .const section.
438  *
439  * Note: None
440  ********************************************************************/
441 uint32_t ReadProgramMemory(uint32_t address)
442 {
443  uint32_t_VAL dwvResult;
444  uint16_t wTBLPAGSave;
445 
446  wTBLPAGSave = TBLPAG;
447  TBLPAG = (uint16_t)(address>>16);
448 
449  dwvResult.w[1] = __builtin_tblrdh((uint16_t)address);
450  dwvResult.w[0] = __builtin_tblrdl((uint16_t)address);
451  TBLPAG = wTBLPAGSave;
452 
453  return dwvResult.Val;
454 }
