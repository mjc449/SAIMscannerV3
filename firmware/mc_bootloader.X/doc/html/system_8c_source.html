72 /*********************************************************************
73  * Function: void SYSTEM_Initialize( SYSTEM_STATE state )
74  *
75  * Overview: Initializes the system.
76  *
77  * PreCondition: None
78  *
79  * Input: SYSTEM_STATE - the state to initialize the system into
80  *
81  * Output: None
82  *
83  ********************************************************************/
84 void SYSTEM_Initialize(SYSTEM_STATE state) {
85  PIN_MANAGER_Initialize();
86  OSCILLATOR_Initialize();
87 
88  switch (state) {
89  case SYSTEM_STATE_USB_START:
90  //Switch to alternate interrupt vector table for bootloader
91  INTCON2bits.ALTIVT = 1;
92 
93  //To enter the bootloader via hardware we jump A14 to C13 and
94  //toggle the pins a few times
95  bool gotoBootloader = true;
96  int i;
97  for (i = 0; i < 16; i++) {
98  _LATC13 = i % 2;
99  _LATC14 = ~(i % 2);
100  Nop();
101  if ((_RA14 != _LATC13) || (_RA15 != _LATC14)) {
102  gotoBootloader = false;
103  break;
104  }
105  }
106 
107  //Otherwise the bootloader can be entered by setting a flag in
108  //program memory (last word == 0xFFFFFE)
109  //Next check for the software flag
110  uint16_t keyVal;
111  TBLPAG = 0x0002;
112  keyVal = __builtin_tblrdl(0xA7FE);
113 
114  if (keyVal == 0xFFFE) {
115  gotoBootloader = true;
116 
117  _erase_flash(0x02A7FE);
118  _write_flash_word24(0x02A7FE, 0xFFFFFF);
119 
120  }
121 
122  if ((gotoBootloader == false) && ((RCON & 0x83) != 0)) {
123  //Switch to app standare IVT for non boot mode
124  INTCON2bits.ALTIVT = 0;
125  __asm__("goto 0x1800");
126  }
127 
128  //Make sure that the general purpose output driver multiplexed with
129  //the VBUS pin is always consistently configured to be tri-stated in
130  //USB applications, so as to avoid any possible contention with the host.
131  //(ex: maintain TRISFbits.TRISF7 = 1 at all times).
132  TRISFbits.TRISF7 = 1;
133 
134  break;
135 
136  case SYSTEM_STATE_USB_SUSPEND:
137  break;
138 
139  case SYSTEM_STATE_USB_RESUME:
140  break;
141  }
142 }

144 void OSCILLATOR_Initialize(void) {
145  // CPDIV 1:1; RCDIV FRC/2; DOZE 1:8; DOZEN disabled; ROI disabled;
146  CLKDIV = 0x3100;
147  // TUN Center frequency;
148  OSCTUN = 0x0000;
149  // CF no clock failure; NOSC PRIPLL; SOSCEN disabled; POSCEN enabled; CLKLOCK unlocked; OSWEN Switch is Complete;
150  __builtin_write_OSCCONH((uint8_t) ((0x0304 >> _OSCCON_NOSC_POSITION) & 0x00FF));
151  __builtin_write_OSCCONL((uint8_t) ((0x0304 | _OSCCON_OSWEN_MASK) & 0xFF));
152  // Wait for Clock switch to occur
153  while (OSCCONbits.OSWEN != 0);
154 }

232 inline static void use_failsafe_stack(void) {
233  static uint8_t failsafe_stack[32];
234  asm volatile (
235  " mov %[pstack], W15\n"
236  :
237  : [pstack]"r"(failsafe_stack)
238  );
239  /* Controls where the stack pointer limit is, relative to the end of the
240  * failsafe stack
241  */
242  SPLIM = (uint16_t) (((uint8_t *) failsafe_stack) + sizeof (failsafe_stack)
243  - FAILSAFE_STACK_GUARDSIZE);
244 }
