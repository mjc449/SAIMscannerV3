36  USBDeviceInit();

40  while(1)
41  {
42  //In case the application uses config bit settings that enable the WDT, we
43  //need to periodically clear it in order for the bootloader to run normally.
44  ClrWdt();
45 
46  SYSTEM_Tasks();
47 
48  #if defined(USB_POLLING)
49  // Interrupt or polling method. If using polling, must call
50  // this function periodically. This function will take care
51  // of processing and responding to SETUP transactions
52  // (such as during the enumeration process when you first
53  // plug in). USB hosts require that USB devices should accept
54  // and process SETUP packets in a timely fashion. Therefore,
55  // when using polling, this function should be called
56  // regularly (such as once every 1.8ms or faster** [see
57  // inline code comments in usb_device.c for explanation when
58  // "or faster" applies]) In most cases, the USBDeviceTasks()
59  // function does not take very long to execute (ex: <100
60  // instruction cycles) before it returns.
61  USBDeviceTasks();
62  #endif
63 
64  /* If the USB device isn't configured yet, we can't really do anything
65  * else since we don't have a host to talk to. So jump back to the
66  * top of the while loop. */
67  if( USBGetDeviceState() < CONFIGURED_STATE )
68  {
69  /* Jump back to the top of the while loop. */
70  continue;
71  }
72 
73  /* If we are currently suspended, then we need to see if we need to
74  * issue a remote wakeup. In either case, we shouldn't process any
75  * keyboard commands since we aren't currently communicating to the host
76  * thus just continue back to the start of the while loop. */
77  if( USBIsDeviceSuspended() == true )
78  {
79  /* Jump back to the top of the while loop. */
80  continue;
81  }
82 
83  //Application specific tasks
84  APP_HIDBootLoaderTasks();

86  }//end while

90 bool USER_USB_CALLBACK_EVENT_HANDLER(int event, void *pdata, uint16_t size)
91 {
92  switch( event )
93  {
94  case EVENT_TRANSFER:
95  break;
96 
97  case EVENT_SOF:
98  /* We are using the SOF as a timer to time the LED indicator. Call
99  * the LED update function here. */
100  APP_LEDUpdateUSBStatus();
101  break;
102 
103  case EVENT_SUSPEND:
104  /* Update the LED status for the suspend event. */
105  APP_LEDUpdateUSBStatus();
106  break;
107 
108  case EVENT_RESUME:
109  /* Update the LED status for the resume event. */
110  APP_LEDUpdateUSBStatus();
111  break;
112 
113  case EVENT_CONFIGURED:
114  /* When the device is configured, we can (re)initialize the demo
115  * code. */
116  APP_HIDBootLoaderInitialize();
117  break;
118 
119  case EVENT_SET_DESCRIPTOR:
120  break;
121 
122  case EVENT_EP0_REQUEST:
123  /* We have received a non-standard USB request. The HID driver
124  * needs to check to see if the request was for it. */
125  USBCheckHIDRequest();
126  break;
127 
128  case EVENT_BUS_ERROR:
129  break;
130 
131  case EVENT_TRANSFER_TERMINATED:
132  break;
133 
134  default:
135  break;
136  }
137  return true;
138 }
