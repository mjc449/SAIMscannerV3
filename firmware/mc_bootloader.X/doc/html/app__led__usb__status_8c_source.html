57 void APP_LEDUpdateUSBStatus(void)
58 {
59  static uint16_t ledCount = 0;
60 
61  if(USBIsDeviceSuspended() == true)
62  {
63  LED_USB_DEVICE_STATE = 0;
64  return;
65  }
66 
67  switch(USBGetDeviceState())
68  {
69  case CONFIGURED_STATE:
70  /* We are configured. Blink fast.
71  * On for 75ms, off for 75ms, then reset/repeat. */
72  if(ledCount == 1)
73  {
74  LED_USB_DEVICE_STATE = 1;
75  }
76  else if(ledCount == 75)
77  {
78  LED_USB_DEVICE_STATE = 0;
79  }
80  else if(ledCount > 150)
81  {
82  ledCount = 0;
83  }
84  break;
85 
86  default:
87  /* We aren't configured yet, but we aren't suspended so let's blink with
88  * a slow pulse. On for 50ms, then off for 950ms, then reset/repeat. */
89  if(ledCount == 1)
90  {
91  LED_USB_DEVICE_STATE = 1;
92  }
93  else if(ledCount == 50)
94  {
95  LED_USB_DEVICE_STATE = 0;
96  }
97  else if(ledCount > 950)
98  {
99  ledCount = 0;
100  }
101  break;
102  }
103 
104  /* Increment the millisecond counter. */
105  ledCount++;
106 }
